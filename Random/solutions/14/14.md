**Time Complexity:** *O(logn),* **Space Complexity:** *O(1).*

```cpp
    class Solution {
    public:
        int searchInsert(vector<int>& nums, int target) {
            int l, m, r;
            l = -1;
            r = nums.size();
            while(r - l > 1) {
                m = l + ((r - l) >> 1);
                if(nums[m] < target) {
                    l = m;
                } else {
                    r = m;
                }
            }
            return r;
        }
    };
```
