**Time Complexity:** *O(n.4<sup>n</sup>),* **Space Complexity:** *O(n.4<sup>n</sup>).*

```cpp
class Solution {
public:
    vector<string> letterCombinations(string digits) {
        if(digits.empty()) return {};
        unordered_map<char, string> digitToLetters {
                           {'2', "abc"}, {'3', "def" },
            {'4', "ghi" }, {'5', "jkl"}, {'6', "mno" },
            {'7', "pqrs"}, {'8', "tuv"}, {'9', "wxyz"}
        };
        vector<string> lettersForDigit(digits.size());
        for(int i=0;i<digits.size();i++) lettersForDigit[i]=digitToLetters[digits[i]];
        int lastIndex =lettersForDigit.size()-1;
        vector<string> result=generateCombinations(lettersForDigit[lastIndex ], lastIndex, lettersForDigit);
        return result;
    }
    vector<string> generateCombinations(const string &letters, int index, vector<string> &lettersForDigit) {
        vector<string> combinations;
        if(index<0) return combinations;
        if(index==0) {
            transform(letters.begin(), letters.end(), back_inserter(combinations), [](char c){ return string(1, c);});
            return combinations;
        }
        vector<string> previousCombos=generateCombinations(lettersForDigit[index-1], index-1, lettersForDigit);
        for(char letter: letters) {
            for(const string &prefix: previousCombos) {
                combinations.push_back(prefix+letter);
            }
        }
        return combinations;
    }
};
```
