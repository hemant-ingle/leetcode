**Time Complexity:** *O(n),* **Space Complexity:** *O(n).*
```cpp
class Solution {
public:
    string convert(string s, int numRows) {
        if(numRows==1) return s;
        vector<string> rows(numRows);
        int pos=0;
        bool goingDown=false;
        for(char c: s) {
	    // cout<<pos<<" "; // shows the pattern of row indices we are generating.
            rows[pos]+=c;
            if(pos==0 || pos==numRows-1) goingDown=!goingDown;
            pos += goingDown ? 1 : -1;
        }
        string result;
        result.reserve(s.size());
        for(string &row: rows) result+=row;
        return result;
    }
};
```
