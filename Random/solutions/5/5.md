**Time Complexity:** *O(1),* **Space Complexity:** *O(1).*
```cpp
class Solution {
public:
    bool isValidSudoku(vector<vector<char>>& board) {
        vector<unordered_set<char>> rowSets(9);
        vector<unordered_set<char>> colSets(9);
        vector<unordered_set<char>> boxSets(9);
        for(int rowIndex=0;rowIndex<9;++rowIndex) {
            for(int columnIndex=0;columnIndex<9;++columnIndex) {
                char cellValue=board[rowIndex][columnIndex];
                if(cellValue=='.') continue;
                int boxIndex = 3 * (rowIndex/3) + (columnIndex/3);
                if(rowSets[rowIndex].count(cellValue) || 
                   colSets[columnIndex].count(cellValue) || 
                   boxSets[boxIndex].count(cellValue)) {
                    return false;
                }
                else {
                    rowSets[rowIndex].insert(cellValue);
                    colSets[columnIndex].insert(cellValue);
                    boxSets[boxIndex].insert(cellValue);
                }
            }
        }
        return true;
    }
};
```
