**Time Complexity:** *get(key):* O(1) *& put(key, value):* O(1) **Space Complexity:** *O(n).*

```cpp
class LRUCache {
private:
    int capacity;
    list<pair<int, int>> cacheList;
    unordered_map<int, list<pair<int, int>>::iterator> cacheMap;
public:
    LRUCache(int capacity): capacity(capacity) {}
    
    int get(int key) {
        auto itr=cacheMap.find(key);
        if(itr==cacheMap.end()) return -1;
        int value=itr->second->second;
        cacheList.splice(cacheList.begin(), cacheList, itr->second);
        return value;
    }
    
    void put(int key, int value) {
        auto itr=cacheMap.find(key);
        if(itr!=cacheMap.end()) {
            itr->second->second=value;
            cacheList.splice(cacheList.begin(), cacheList, itr->second);
        }
        else {
            if (cacheList.size()==capacity) {
                auto [oldKey, oldValue]=cacheList.back();
                cacheList.pop_back();
                cacheMap.erase(oldKey);
            }
            cacheList.emplace_front(key, value);
            cacheMap[key]=cacheList.begin();
        }
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */
```
